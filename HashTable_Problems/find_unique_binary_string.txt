Problem Name: Find Unique Binary String

Solution Code:

class Solution {
    public String findDifferentBinaryString(String[] nums) {
        
        int n = nums.length;

        HashSet<String> set = new HashSet<>();

        for(int i=0; i<n; i++){
            set.add(nums[i]);
        }

        String ans = checkValidString("", set, n);

        return ans;
    }

    public static String checkValidString(String curr, HashSet<String> set, int n){
        if(curr.length() == n){
            if(!set.contains(curr)){
                System.out.println(curr);
                return curr;
            }
            return "";
        }

        String addZero = checkValidString(curr + '0', set, n);
        System.out.println(curr.length());
        if(addZero.length() > 0){
            return addZero;
        }
        return checkValidString(curr + '1', set, n);
    }
}
