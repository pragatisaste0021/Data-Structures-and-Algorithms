Problem Name: Stock Price Fluctuation 

Solution Code:

class StockPrice {

    TreeMap<Integer, Integer> map1;
    TreeMap<Integer, Set<Integer>> map2;

    public StockPrice() {
        map1 = new TreeMap<>();
        map2 = new TreeMap<>();
    }
    
    public void update(int timestamp, int price) {
        if(map1.containsKey(timestamp)){
            int prevPrice = map1.get(timestamp);
            Set<Integer> set = map2.get(prevPrice);
            set.remove(timestamp);
            if(set.isEmpty()){
                map2.remove(prevPrice);
            }
        }
        map2.putIfAbsent(price,new HashSet<>());
        map2.get(price).add(timestamp);
        map1.put(timestamp, price);
    }
    
    public int current() {
        return map1.lastEntry().getValue();
    }
    
    public int maximum() {
        return map2.lastKey();
    }
    
    public int minimum() {
        return map2.firstKey();
    }
}
