class Solution {
    public boolean isValidSudoku(char[][] board) {
        
        HashSet<Character> set1 = new HashSet<>();
        HashSet<Character> set2 = new HashSet<>();
        HashSet<Character> set3 = new HashSet<>();

        for(int i=0; i<9; i++){
            if(i == 3 || i == 6){
                
                set1.clear();
                set2.clear();
                set3.clear();
            }
            for(int j=0; j<9; j++){
                if(j < 3 && board[i][j] >=48 && board[i][j] <= 58){
                    if(set1.contains(board[i][j])){
                        return false;
                    }
                    set1.add(board[i][j]);
                }
                else if(j >= 3 && j < 6 && board[i][j] >= 48 && board[i][j] <= 58){
                    if(set2.contains(board[i][j])){
                        return false;
                    }
                    set2.add(board[i][j]);
                }
                else if(j >= 6 && j < 9 && board[i][j] >= 48 && board[i][j] <= 58){
                    if(set3.contains(board[i][j])){
                        return false;
                    }
                    set3.add(board[i][j]);
                }
            }
        }

        System.out.println("jfhdf");

        // Rows

        for(int i=0; i<9; i++){
            set1.clear();
            for(int j=0; j<9; j++){
                if(board[i][j] >= 48 && board[i][j] <= 58 && set1.contains(board[i][j])){
                    return false;
                }
                if(board[i][j] >= 48 && board[i][j] <= 58){
                    set1.add(board[i][j]);
                }
            }
        }

        // Columns
        for(int i=0; i<9; i++){
            set1.clear();
            for(int j=0; j<9; j++){
                if(board[j][i] >= 48 && board[j][i] <= 58 && set1.contains(board[j][i])){
                    return false;
                }
                if(board[j][i] >= 48 && board[j][i] <= 58){
                    set1.add(board[j][i]);
                }
            }
        }
        return true;
    }
}
