Problem Name: Subdomain Visit Count

Solution Code:

class Solution {
    public List<String> subdomainVisits(String[] cpdomains) {
        
        int n = cpdomains.length;

        List<String> list = new ArrayList<>();

        HashMap<String, Integer> map = new HashMap<>();

        for(int i=0; i<n; i++){
            int j = 0;

            while(j < cpdomains[i].length() && cpdomains[i].charAt(j) != ' '){
                j++;
            }
            int value = Integer.parseInt(cpdomains[i].substring(0, j));

            String str = cpdomains[i].substring(j+1, cpdomains[i].length());

            // System.out.println(str);

            // System.out.println(map.getOrDefault(str, 0));

            map.put(str, map.getOrDefault(str, 0) + value);

            while(j < cpdomains[i].length()){
                if(cpdomains[i].charAt(j) == '.'){
                    str = cpdomains[i].substring(j+1, cpdomains[i].length());
                    map.put(str, map.getOrDefault(str, 0) + value);
                }
                j++;
            }
        }

        for(Map.Entry<String, Integer> entry : map.entrySet()){

            StringBuilder str = new StringBuilder();

            String key = entry.getKey();
            int value = entry.getValue();

            String strValue = String.valueOf(value);

            str.append(strValue).append(' ').append(key);

            list.add(str.toString());
        }
        return list;  
    }
}
