Problem Name: Delete Nodes And Return Forest

Solution Code:

class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        
        if(root == null){
            return new ArrayList<>();
        }

        Queue<TreeNode> queue = new LinkedList<>();

        queue.add(root);

        int n = to_delete.length;

        HashSet<Integer> set = new HashSet<>();

        for(int i=0; i<n; i++){
            set.add(to_delete[i]);
        }

        List<TreeNode> list = new ArrayList<>();

        while(!queue.isEmpty()){

            TreeNode currNode = queue.poll();

            if(currNode.left != null){
                queue.add(currNode.left);

                if(set.contains(currNode.left.val)){
                    currNode.left = null;
                }
            }

            if(currNode.right != null){
                queue.add(currNode.right);

                if(set.contains(currNode.right.val)){
                    currNode.right = null;
                }
            }

            if(set.contains(currNode.val)){
                if(currNode.left != null){
                    list.add(currNode.left);
                }
                if(currNode.right != null){
                    list.add(currNode.right);
                }
            }
        }

        if(!set.contains(root.val)){
            list.add(root);
        }

        return list;  
    }
}
