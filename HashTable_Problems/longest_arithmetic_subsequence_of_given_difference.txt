Problem Name: Longest Arithmetic Subsequence of Given Difference

Solution Code:

class Solution {
    public int longestSubsequence(int[] arr, int difference) {
        
        int m = arr.length;

        HashMap<Integer, Integer> map = new HashMap<>();

        int maxLen = 0;

        for(int i=0; i<m; i++){

            int val1 = arr[i] - difference;
            // int val2 = arr[i] + difference;

            int len = map.getOrDefault(val1, 0) + 1;

            map.put(arr[i], len);

            // System.out.println(map);

            maxLen = Math.max(maxLen, len);

            // System.out.println(maxLen + " " + arr[i]);
        }

        return maxLen;
       
    }
}
