Problem Name: Split a String Into the Max Number of Unique Substrings

Solution Code:

class Solution {
    public int maxUniqueSplit(String s) {
        
        int n = s.length();

        Set<String> set = new HashSet<>();

        int[] maxCount = new int[1];

        backTrack(s, 0, set, 0, maxCount);

        return maxCount[0];
    }

    public static void backTrack(String s, int start, Set<String> seen, int count, int[] maxCount){

        if(count + (s.length() - start) <= maxCount[0]) return;

        if(start == s.length()){
            maxCount[0] = Math.max(maxCount[0], count);
            return;
        }

        for(int end=start+1; end <= s.length(); end++){

            String substring = s.substring(start, end);

            if(!seen.contains(substring)){
                seen.add(substring);
                backTrack(s, end, seen, count+1, maxCount);
                seen.remove(substring);
            }
        }
    }
}
