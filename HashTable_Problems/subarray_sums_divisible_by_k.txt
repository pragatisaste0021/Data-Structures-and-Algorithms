Problem Name: Subarray Sums Divisible by K

Solution Code:

class Solution {
    public int subarraysDivByK(int[] nums, int k) {
        int count = 0;
        int n = nums.length;

        HashMap<Integer, Integer> map = new HashMap<>();
        
        map.put(0, 1);
        int sum = 0;
        
        for(int i=0; i<n; i++){
        sum += nums[i];
        int rem = ((sum % k) + k)%k;
        count += map.getOrDefault(rem, 0);
        map.put(rem, map.getOrDefault(rem, 0) + 1);
        }
        return count;
    }
}
