Problem Name: Letter Tile Possibilities

Solution Code:

class Solution {
    public int numTilePossibilities(String tiles) {
        
        int n = tiles.length();

        HashSet<String> set = new HashSet<>();

        char[] chars = tiles.toCharArray();
        Arrays.sort(chars);
        String sortedTiles = new String(chars);

        return generateSequences(sortedTiles, set, "", 0) - 1;
    }

    public static int generateSequences(String tiles, HashSet<String> set, String seq, int index){
        if(index >= tiles.length()){
            if(set.add(seq)){
                return countPermutations(seq);
            }
            return 0;
        }

        return generateSequences(tiles, set, seq, index+1) + generateSequences(tiles, set, seq + tiles.charAt(index), index+1);
    }

    public static int countPermutations(String seq){
        int[] countChar = new int[26];

        for(char ch : seq.toCharArray()){
            countChar[ch - 'A']++;
        }

        int total = findFactorial(seq.length());

        for(int count : countChar){
            total = total / findFactorial(count);
        }
        return total;
    }

    public static int findFactorial(int num){
        if(num <= 1){
            return 1;
        }
        int result = 1;
        for(int i=2; i<=num; i++){
            result *= i;
        }
        return result;
    }
}
