Problem Name: Check If Array Pairs Are Divisible by k

Solution Code:

class Solution {
    public boolean canArrange(int[] arr, int k) {
        
        Map<Integer, Integer> map = new HashMap<>();

        for (int i : arr) {
            int rem = ((i % k) + k) % k;
            map.put(rem, map.getOrDefault(rem, 0) + 1);
        }

        for (int i : arr) {
            int rem = ((i % k) + k) % k;

            if (rem == 0) {
                if (map.get(rem) % 2 == 1) return false;
            }
           
            else if (
                !Objects.equals(
                    map.get(rem),
                    map.get(k - rem)
                )
            ) return false;
        }
        return true;
    }
}
