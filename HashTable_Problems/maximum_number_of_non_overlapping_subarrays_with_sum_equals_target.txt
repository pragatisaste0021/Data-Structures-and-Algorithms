Problem Name: Maximum Number of Non-Overlapping Subarrays With Sum Equals Target

Solution Code:

class Solution {
    public int maxNonOverlapping(int[] nums, int target) {
        
        int n = nums.length;

        HashMap<Integer, Integer> map = new HashMap<>();

        int totalSubarrays = 0;

        int prefixSum = 0;

        map.put(0, -1);

        int availableIndex = -1;

        for(int i=0; i<n; i++){
            prefixSum += nums[i];
            if(map.containsKey(prefixSum - target) && map.get(prefixSum - target) >= availableIndex){
                totalSubarrays++;
                availableIndex = i;
            }
            map.put(prefixSum, i);
        }
        return totalSubarrays;
    }
}
