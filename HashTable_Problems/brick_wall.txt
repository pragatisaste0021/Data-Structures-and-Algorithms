Problem Name: Brick Wall

Solution Code: 

class Solution {
    public int leastBricks(List<List<Integer>> wall) {
        
        HashMap<Integer, Integer> map = new HashMap<>();

        int totalWidth = 0;

        for(List<Integer> li : wall){
            int sum = 0;
            for(int i=li.size()-1; i>=0; i--){
                sum += li.get(i);
                map.put(sum, map.getOrDefault(sum, 0) + 1);
            }
            totalWidth = sum;
        }

        int commonSum = -1;

        for(Map.Entry<Integer, Integer> entry : map.entrySet()){
            if(entry.getKey() != totalWidth){
                commonSum = Math.max(commonSum, entry.getValue());
            }
        }

        if(commonSum == -1){
            return wall.size();
        }
        return wall.size() - commonSum;
    }
}
